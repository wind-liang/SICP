(define (gcd b r)
    (if (= 0 r)
        b
        (gcd r (remainder b r))))
        
(define (make-rat n d)  
    (define g (gcd n d)) 
    (let ((a (/ n g))
          (b (/ d g)))
        (if (< b 0)
            (cons (- a) (- b))
            (cons a b))))
        
(define (number x) (car x))
(define (denom x) (cdr x))
(define (print-rat x)
    (newline)
    (display (number x))
    (display "/")
    (display (denom x)))
